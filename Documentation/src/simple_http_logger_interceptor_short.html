class interface
	SIMPLE_HTTP_LOGGER_INTERCEPTOR

create 
	make,
	make_verbose

feature -- Configuration

	set_verbose (a_verbose: BOOLEAN)
			-- Set verbose mode.
		ensure
			verbose_set: verbose = a_verbose

	set_prefix (a_prefix: STRING_8)
			-- Set log prefix.
		require
			prefix_not_void: a_prefix /= Void
		ensure
			prefix_set: log_prefix = a_prefix
	
feature -- Interception

	before_request (a_url, a_method: STRING_8; a_headers: HASH_TABLE [STRING_8, STRING_8]; a_body: detachable STRING_8)
			-- Log request details.

	after_response (a_url, a_method: STRING_8; a_response: SIMPLE_HTTP_RESPONSE)
			-- Log response details.

	on_error (a_url, a_method: STRING_8; a_error_message: STRING_8)
			-- Log error.
