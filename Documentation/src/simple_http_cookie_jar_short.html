	cookie_header: STRING_8
			-- Build Cookie header value from stored cookies.

	set_cookie (a_name, a_value: STRING_8)
			-- Set a cookie.
		require
			name_not_empty: not a_name.is_empty
		ensure
			cookie_set: cookies.has (a_name)

	remove_cookie (a_name: STRING_8)
			-- Remove a cookie.
		require
			name_not_empty: not a_name.is_empty
		ensure

		ensure
